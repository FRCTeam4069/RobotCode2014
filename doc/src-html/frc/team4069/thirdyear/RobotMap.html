003/**
004 * The RobotMap is a mapping from the ports sensors and actuators are wired into
005 * to a variable name. This provides flexibility changing wiring, makes checking
006 * the wiring easier and significantly reduces the number of magic numbers
007 * floating around.
008 */
009public interface RobotMap {
010
011    //DRIVETRAIN
012    /**
013     * Left drive motor port.
014     */
015    public static final int LEFT_MOTOR = 1;
016    /**
017     * Right drive motor port.
018     */
019    public static final int RIGHT_MOTOR = 2;
020    /**
021     * Left encoder Digital I/O ports.
022     */
023    public static final int LEFT_ENCODER_1 = 2;
024    public static final int LEFT_ENCODER_2 = 3;
025    /**
026     * Right encoder Digital I/O ports.
027     */
028    public static final int RIGHT_ENCODER_1 = 4;
029    public static final int RIGHT_ENCODER_2 = 5;
030    //SHOOTER
031    /**
032     * Winch forward and backward solenoid ports.
033     */
034    public static final int WINCH_PISTON_1 = 3;
035    public static final int WINCH_PISTON_2 = 4;
036    /**
037     * Winch motor port.
038     */
039    public static final int WINCH_MOTOR = 4;
040    /**
041     * Compressor relay port.
042     */
043    public static final int COMPRESSOR = 1;
044    /**
045     * Pressure switch port.
046     */
047    public static final int PRESSURE_SWITCH = 1;
048    /**
049     * Arm potentiometer port.
050     */
051    public static final int ARM_POTENTIOMETER = 2;
052    //PICKUP    
053    /**
054     * Pickup arm forward and backward solenoid ports.
055     */
056    public static final int PICKUP_PISTON_1 = 1;
057    public static final int PICKUP_PISTON_2 = 2;
058    /**
059     * Pickup roller port.
060     */
061    public static final int PICKUP_ROLLER = 5;
062    /**
063     * Reed switch port.
064     */
065    public static final int WINCH_REEDSWITCH = 6;
066}
