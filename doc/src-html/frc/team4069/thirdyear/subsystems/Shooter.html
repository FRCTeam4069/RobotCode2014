001package frc.team4069.thirdyear.subsystems;
002
003import edu.wpi.first.wpilibj.AnalogChannel;
004import edu.wpi.first.wpilibj.Compressor;
005import edu.wpi.first.wpilibj.DigitalInput;
006import edu.wpi.first.wpilibj.DoubleSolenoid;
007import edu.wpi.first.wpilibj.Talon;
008import edu.wpi.first.wpilibj.command.Subsystem;
009import frc.team4069.thirdyear.RobotMap;
010
011/**
012 * Shooter subsystem incorporating winch, transmission, and safety features.
013 *
014 * @author Edmund
015 */
016public class Shooter extends Subsystem {
017    // Put methods for controlling this subsystem
018    // here. Call these from Commands.
019
020    DoubleSolenoid transmissionSolenoid;
021    Talon winchTalon;
022    AnalogChannel armPotentiometer;
023    DigitalInput reedSwitch;
024    Compressor compressor;
025    private static final double SHOOTING_ANGLE = 127.73;
026    private static final double WINCH_P = 0.075;
027
028    public Shooter() {
029        transmissionSolenoid = new DoubleSolenoid(RobotMap.WINCH_PISTON_1,
030                RobotMap.WINCH_PISTON_2);
031        winchTalon = new Talon(RobotMap.WINCH_MOTOR);
032        armPotentiometer = new AnalogChannel(RobotMap.ARM_POTENTIOMETER);
033        armPotentiometer.setOversampleBits(5);
034        compressor = new Compressor(RobotMap.PRESSURE_SWITCH,
035                RobotMap.COMPRESSOR);
036        compressor.start();
037        reedSwitch = new DigitalInput(RobotMap.WINCH_REEDSWITCH);
038    }
039
040    /**
041     * Makes the shooty thing go pow.
042     *
043     * @param forward If true, disengages the transmission and fires. If false,
044     *                re-engages the transmission.
045     */
046    public void fireSolenoid(boolean forward) {
047        transmissionSolenoid.set(forward ? DoubleSolenoid.Value.kReverse :
048                DoubleSolenoid.Value.kForward);
049    }
050    private static final double MAX_ANGLE = 135;
051    private static final double MIN_ANGLE = 96;
052
053    /**
054     * By convention, positive speed tightens the winch,
055     * negative speed releases it.
056     * Don't do anything else.
057     *
058     * @param speed
059     */
060    public void spinWinch(double speed) {
061        if ((getPotentiometerAngle() < MAX_ANGLE && speed >= 0)
062                || (getPotentiometerAngle() > MIN_ANGLE && speed <= 0)) {
063            winchTalon.set(speed);
064        } else {
065            winchTalon.set(0);
066        }
067//        if (!reedSwitch.get()) {
068//            winchTalon.set(speed);
069//        } else {
070//            winchTalon.set(0);
071//        }
072    }
073    private static final double POT_VALUE_AT_ZERO_VOLTS = -547.0336620966;
074    private static final double POT_VALUE_SLOPE = 366.3256030533;
075
076    /**
077     * Gets the angle of the shooter arm potentiometer from the vertical. Scales
078     * the voltage according to the constants above. DO NOT change these
079     * constants; angular constants such as {@link Shooter#SHOOTING_ANGLE }
080     * depend on them.
081     *
082     * @return The angle in degrees of the shooter arm.
083     */
084    public double getPotentiometerAngle() {
085        double voltage = armPotentiometer.getAverageVoltage();
086        double value = POT_VALUE_AT_ZERO_VOLTS + voltage * POT_VALUE_SLOPE;
087        return value;
088    }
089
090    /**
091     * The default command is ReloadCommand. May be unwise - check with team?
092     */
093    public void initDefaultCommand() {
094        // Set the default command for a subsystem here.
095        //setDefaultCommand(new MySpecialCommand());
096    }
097
098    /**
099     * Checks if the pickup arm is in the correct position for shooter usage.
100     *
101     * @return The value of the reed switch on the pickup arm.
102     */
103    public boolean getReedSwitch() {
104        return reedSwitch.get();
105    }
106
107    /**
108     * Checks if the shooter arm is below or at the maximum angle.
109     *
110     * @return If the arm is ready to receive a ball.
111     */
112    public boolean isArmReady() {
113        return getPotentiometerAngle() >= MAX_ANGLE;
114    }
115
116    /**
117     * Moves the winch to the shooting angle using proportional control. Makes
118     * use of {@link Shooter#moveToAngle(double)}.
119     */
120    public void moveToShootingAngle() {
121        moveToAngle(SHOOTING_ANGLE);
122    }
123
124    /**
125     * Moves the winch to the desired angle using proportional control. Uses
126     * {@link Shooter#getPotentiometerAngle()}.
127     *
128     * @param The desired angle measured from the vertical.
129     */
130    public void moveToAngle(double angle) {
131        spinWinch(WINCH_P * (angle - getPotentiometerAngle()));
132    }
133}
