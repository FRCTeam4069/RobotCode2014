005package frc.team4069.thirdyear.subsystems;
006
007import edu.wpi.first.wpilibj.DoubleSolenoid;
008import edu.wpi.first.wpilibj.Talon;
009import edu.wpi.first.wpilibj.command.Subsystem;
010import frc.team4069.thirdyear.RobotMap;
011
012/**
013 * Pickup system. Includes roller motor and arm solenoid.
014 * Batteries not included.
015 *
016 * @author Edmund
017 */
018public class Pickup extends Subsystem {
019
020    DoubleSolenoid pickupArm;
021    Talon pickupRoller;
022
023    public Pickup() {
024        pickupArm = new DoubleSolenoid(RobotMap.PICKUP_PISTON_1,
025                RobotMap.PICKUP_PISTON_2);
026        pickupRoller = new Talon(RobotMap.PICKUP_ROLLER);
027    }
028
029    /**
030     * Makes the pickup mechanism move.
031     *
032     * @param up The direction of movement. True for upward, false for
033     *           downward.
034     */
035    public void move(boolean up) {
036        pickupArm.set(up ? DoubleSolenoid.Value.kReverse :
037                DoubleSolenoid.Value.kForward);
038    }
039
040    /**
041     * Spins the pickup roller.
042     *
043     * @param speed The speed for spinning. Negative values ignored.
044     */
045    public void spin(double speed) {
046        if (speed < 0) {
047            speed = 0;
048        }
049        pickupRoller.set(speed);
050    }
051
052    /**
053     * There is <b>no</b> default command.
054     */
055    public void initDefaultCommand() {
056        // Set the default command for a subsystem here.
057        //setDefaultCommand(new MySpecialCommand());
058    }
059}
