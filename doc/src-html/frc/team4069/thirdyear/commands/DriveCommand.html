005package frc.team4069.thirdyear.commands;
006
007import com.sun.squawk.util.MathUtils;
008import edu.wpi.first.wpilibj.DigitalInput;
009import edu.wpi.first.wpilibj.DriverStation;
010import edu.wpi.first.wpilibj.Joystick;
011import edu.wpi.first.wpilibj.command.Command;
012import java.util.Date;
013
014/**
015 * The command that runs during operator control.
016 *
017 * @author Edmund
018 */
019public class DriveCommand extends CommandBase {
020
021    private DriverStation m_ds;
022    private Joystick driveStick;
023    private static final int FIR_LENGTH = 5;
024    private double lastLeftStickX;
025    private double lastLeftStickY;
026
027    /**
028     * Constructor.
029     */
030    public DriveCommand() {
031        requires(drivetrain);
032        requires(pickup);
033        requires(shooter);
034        lastLeftStickX = 0;
035        lastLeftStickY = 0;
036        m_ds = DriverStation.getInstance();
037        driveStick = oi.getDriveStick();
038        // Use requires() here to declare subsystem dependencies
039        // eg. requires(chassis);
040    }
041
042    /**
043     * Called at the beginning of operator control.
044     * Makes the drivetrain brake.
045     */
046    protected void initialize() {
047        drivetrain.brake();
048    }
049
050    /**
051     * Periodically called during operator control.
052     */
053    protected void execute() {
054        double currentLeftStickX = (driveStick.getRawAxis(1)) * Math.abs(
055                driveStick.getRawAxis(
056                1)) / 1.8;
057        double currentLeftStickY = driveStick.getRawAxis(3);
058        double outputY = 0;
059        double outputX = 0;
060        outputX = currentLeftStickX * 0.07 + lastLeftStickX * 0.93;
061        outputY = currentLeftStickY * 0.00006 + lastLeftStickY * 0.99994;
062        lastLeftStickX = outputX;
063        lastLeftStickY = outputY;
064        drivetrain.arcadeControlledDrive(currentLeftStickY, currentLeftStickX);
065        boolean pickupIsSafe = !shooter.getReedSwitch();
066        if (pickupIsSafe) {
067            if (driveStick.getRawButton(3)) {
068                shooter.spinWinch(-0.6);
069            } else if (driveStick.getRawButton(2)) {
070                shooter.spinWinch(1);
071            } else if (driveStick.getRawButton(1)) {
072                shooter.spinWinch(0);
073                shooter.fireSolenoid(true);
074
075            } else {
076                shooter.spinWinch(0);
077                shooter.fireSolenoid(false);
078            }
079            if (shooter.isArmReady()) {
080                if (driveStick.getRawButton(4)) {
081                    pickup.spin(1);
082                } else {
083                    pickup.spin(0);
084                }
085                if (driveStick.getRawButton(5)) {
086                    pickup.move(true);
087                } else if (driveStick.getRawButton(6)) {
088                    pickup.move(false);
089                }
090            }
091        }
092        System.out.println(shooter.getPotentiometerAngle());
093    }
094
095    // Make this return true when this Command no longer needs to run execute()
096    protected boolean isFinished() {
097        return !m_ds.isOperatorControl();
098    }
099
100    // Called once after isFinished returns true
101    protected void end() {
102    }
103
104    // Called when another command which requires one or more of the same
105    // subsystems is scheduled to run
106    protected void interrupted() {
107    }
108}
