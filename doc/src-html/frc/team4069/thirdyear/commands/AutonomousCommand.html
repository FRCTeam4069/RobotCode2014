001package frc.team4069.thirdyear.commands;
002
003import edu.wpi.first.wpilibj.DriverStation;
004import edu.wpi.first.wpilibj.Timer;
005import edu.wpi.first.wpilibj.command.Scheduler;
006import edu.wpi.first.wpilibj.networktables.NetworkTable;
007import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
008import java.util.Date;
009
010/**
011 * Command that moves the robot when the target is detected.
012 *
013 * @author Edmund
014 */
015public class AutonomousCommand extends CommandBase {
016
017    protected NetworkTable roborealmTable;
018    private DriverStation m_ds;
019    private boolean hasLockOn;
020    private boolean inPosition;
021    private Date lockonTime;
022    private Date inPositionTime;
023    double blobCountFiltered;
024    private static final double GOAL_DIST = 15 * 12;
025
026    public AutonomousCommand() {
027        requires(drivetrain);
028        requires(shooter);
029        m_ds = DriverStation.getInstance();
030        // Use requires() here to declare subsystem dependencies
031        // eg. requires(chassis);
032    }
033
034    // Called just before this Command runs the first time
035    protected void initialize() {
036        roborealmTable = NetworkTable.getTable("SmartDashboard");
037        drivetrain.brake();
038        blobCountFiltered = 0;
039        hasLockOn = false;
040        inPosition = false;
041        drivetrain.resetDistance();
042        Timer.delay(0.1);
043    }
044
045    /**
046     * Checks for the vision target, drives to the low goal and shoots into the
047     * high goal.
048     * Future implementations may forgo checking for the vision target first,
049     * because if it isn't triggered at the start it will be in the next five
050     * seconds.
051     */
052    protected void execute() {
053        pickup.move(false);
054        double blobCount;
055        blobCount = ((Double) roborealmTable.getValue("BLOB_COUNT",
056                Double.valueOf(0))).doubleValue();
057        blobCountFiltered = blobCount * 0.3 + blobCountFiltered * 0.7;
058        if (blobCountFiltered > 0.5 && !hasLockOn) {
059            hasLockOn = true;
060            lockonTime = new Date();
061        }
062        if (hasLockOn) {
063            driveLoop();
064        } else if (inPosition) {
065            shootLoop();
066        } else {
067            drivetrain.brake();
068        }
069        System.out.println("Speed " + drivetrain.getSpeed());
070    }
071
072    /**
073     * Runs at the beginning of the autonomous loop. Drives the robot to the low
074     * goal.
075     */
076    private void driveLoop() {
077        if (drivetrain.getDistance() < GOAL_DIST * 0.73) {
078            drivetrain.arcadeControlledDrive(-0.7, 0);
079        } else if (drivetrain.getDistance() < GOAL_DIST * 0.91) {
080            drivetrain.arcadeControlledDrive(-0.25, 0);
081        } else if (drivetrain.getDistance() < GOAL_DIST) {
082            shooter.moveToShootingAngle();
083            drivetrain.arcadeControlledDrive(-0.25, 0);
084        }
085        if (drivetrain.getDistance() > GOAL_DIST
086                || (Math.abs(drivetrain.getSpeed()) < 1.0
087                && new Date().getTime() - lockonTime.getTime() > 1000)) {
088            drivetrain.arcadeControlledDrive(0.05, 0);
089            shooter.spinWinch(0);
090            hasLockOn = false;
091            inPosition = true;
092            inPositionTime = new Date();
093        }
094    }
095
096    /**
097     * Runs when the robot has detected that it is at the low goal.
098     */
099    private void shootLoop() {
100        drivetrain.brake();
101        if (new Date().getTime() - inPositionTime.getTime() > 200) {
102            shooter.fireSolenoid(true);
103        }
104    }
105
106    // Make this return true when this Command no longer needs to run execute()
107    protected boolean isFinished() {
108        return !m_ds.isAutonomous();
109    }
110
111    // Called once after isFinished returns true
112    protected void end() {
113    }
114
115    // Called when another command which requires one or more of the same
116    // subsystems is scheduled to run
117    protected void interrupted() {
118    }
119}
