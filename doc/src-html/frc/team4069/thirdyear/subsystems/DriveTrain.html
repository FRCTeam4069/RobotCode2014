001package frc.team4069.thirdyear.subsystems;
002
003import edu.wpi.first.wpilibj.Talon;
004import edu.wpi.first.wpilibj.command.Subsystem;
005import edu.wpi.first.wpilibj.livewindow.LiveWindow;
006import frc.team4069.thirdyear.RobotMap;
007
008/**
009 * Drivetrain subsystem.
010 * @author Edmund
011 */
012public class DriveTrain extends Subsystem {
013
014    private Talon leftMotor;
015    private Talon rightMotor;
016
017    public DriveTrain() {
018        super();
019        leftMotor = new Talon(RobotMap.LEFT_MOTOR);
020        rightMotor = new Talon(RobotMap.RIGHT_MOTOR);
021    }
022    
023    /**
024     * Adds motors to the live window for use during test mode.
025     */
026    
027    private void initLiveWindow() {
028        LiveWindow.addActuator("Drivetrain", "Left Motors", leftMotor);
029        LiveWindow.addActuator("Drivetrain", "Right Motors", rightMotor);
030    }
031
032    public void initDefaultCommand() {
033        // Set the default command for a subsystem here.
034        //setDefaultCommand(new MySpecialCommand());
035    }
036
037    /**
038     * Tank drive: makes the robot move.
039     *
040     * @param left Speed for the left motors.
041     * @param right Speed for the right motors.
042     */
043    public void tankDrive(double left, double right) {
044        leftMotor.set(left);
045        rightMotor.set(right);
046    }
047
048    /**
049     * Stops the robot. May make use of encoder-based PID, for now it just sets
050     * speed to zero.
051     */
052    public void brake() {
053        leftMotor.set(0);
054        rightMotor.set(0);
055    }
056}
