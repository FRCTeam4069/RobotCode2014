021    private static final double DISTANCE_PER_PULSE = (24.0 / 42.0) * 3.0 * 4.0 * Math.PI / 128;
022    //0.00467498906784195422390274312988;//0.05609986881410345068683291755856;//0.0146869113074526041414067839586;//0.17624293568943124969688;//0.0140249672;
023    private static final double MAX_INCH_PER_SECOND = 13 * 12;
024    private Talon leftMotor;
025    private Talon rightMotor;
026    private Encoder leftEncoder;
027    private Encoder rightEncoder;
028    private static final double speedP = 1.2, speedI = 0.009, speedD = 0.02;
029    private static final double distP = 0.1, distI = 0, distD = 0;
030    private PIDController arcadeController;
031    private PIDController distanceController;
032
033    /**
034     * Drivetrain constructor.
035     */
036    public DriveTrain() {
037        super();
038        leftMotor = new Talon(RobotMap.LEFT_MOTOR);
039        rightMotor = new Talon(RobotMap.RIGHT_MOTOR);
040        leftEncoder = new Encoder(RobotMap.LEFT_ENCODER_1,
041                RobotMap.LEFT_ENCODER_2);
042        rightEncoder = new Encoder(RobotMap.RIGHT_ENCODER_1,
043                RobotMap.RIGHT_ENCODER_2);
044        leftEncoder.setDistancePerPulse(DISTANCE_PER_PULSE);
045        rightEncoder.setDistancePerPulse(DISTANCE_PER_PULSE);
046        leftEncoder.setSamplesToAverage(6);
047        rightEncoder.setSamplesToAverage(6);
048        leftEncoder.setReverseDirection(true);
049        leftEncoder.start();
050        rightEncoder.start();
051        arcadeController = new PIDController(speedP, speedI, speedD,
052                encoderArcadeSource, encoderArcadeOutput);
053        arcadeController.setInputRange(-1, 1);
054        arcadeController.setOutputRange(-1, 1);
055        distanceController = new PIDController(distP, distI, distD,
056                encoderDistSource, encoderDistOutput);
057    }
058

063        LiveWindow.addActuator("Drivetrain", "Left Motors", leftMotor);
064        LiveWindow.addActuator("Drivetrain", "Right Motors", rightMotor);
065    }

072    /**
073     * Tank drive: makes the robot move.
074     *
075     * @param left  Speed for the left motors.
076     * @param right Speed for the right motors.
077     */
078    public void tankDrive(double left, double right) {
079        leftMotor.set(-left * 0.8);
080        rightMotor.set(right * 0.8);
081    }
082  
083    double controllerValue;
084  /**
085     * Drives, actively compensating for drift.
086     *
087     * @param moveValue   Speed for forward movement.
088     * @param rotateValue Speed for turning. PID controlled.
089     */
090    public void arcadeControlledDrive(double moveValue, double rotateValue) {
091        arcadeController.enable();
092        arcadeController.setSetpoint(rotateValue);
093        controllerValue = arcadeController.get() * 0.3 + controllerValue * 0.7;
094        arcadeDrive(moveValue, rotateValue + controllerValue);// + arcadeController.get());
095    }
096
097    /**
098     * Arcade drive: drives using a turn value and move value.
099     *
100     * @param moveValue   The speed at which the robot will move forward and
101     *                    backward.
102     * @param rotateValue The speed at which the robot will turn. Right values
103     *                    are right turns.
104     */
105    public void arcadeDrive(double moveValue, double rotateValue) {
106        double leftMotorSpeed;
107        double rightMotorSpeed;
108        double theta = MathUtils.atan2(moveValue, rotateValue);
109        double r = Math.sqrt(
110                (moveValue * moveValue) + (rotateValue * rotateValue));
111        double sinTheta = Math.sin(theta);
112        double cosTheta = Math.cos(theta);
113        leftMotorSpeed = (sinTheta + cosTheta) * r;
114        rightMotorSpeed = (sinTheta - cosTheta) * r;
115        //Log.log("Left: " + leftMotorSpeed + " Right: " + rightMotorSpeed);
116        tankDrive(leftMotorSpeed, rightMotorSpeed);
117    }
118
119    /**
120     * Stops the robot. May make use of encoder-based PID, for now it just sets
121     * speed to zero.
122     */
123    public void brake() {
124        leftMotor.set(0);
125        rightMotor.set(0);
126    }
127
128    /**
129     * Gets the robot's speed. Only really works when not turning.
130     *
131     * @return Speed in meters / second.
132     */
133    public double getSpeed() {
134        double leftSpeed = leftEncoder.getRate();
135        double rightSpeed = rightEncoder.getRate();
136        double averageSpeed = leftSpeed / 2.0 + rightSpeed / 2.0;
137        return averageSpeed;
138    }
139    /**
140     * Converts encoder speeds to an arcade drive joystick x-value.
141     *
142     * @return
143     */
144    private double lastTurnSpeed = 0;
145
146    protected double getTurnValue() {
147        double leftSpeed = leftEncoder.getRate() / MAX_INCH_PER_SECOND * 3;
148        double rightSpeed = rightEncoder.getRate() / MAX_INCH_PER_SECOND * 3;
149        double forwardSpeed = leftSpeed / 2.0 + rightSpeed / 2.0;
150        double turnSpeed = (leftSpeed / 2.0 - rightSpeed / 2.0); //* 0.1 + lastTurnSpeed * 0.9;
151        lastTurnSpeed = turnSpeed;
152        return turnSpeed;
153    }
154
155    /**
156     * Gets the robot's distance travelled since last reset. Only really works
157     * when not turning.
158     *
159     * @return Distance in meters.
160     */
161    public double getDistance() {
162        double leftDistance = leftEncoder.getDistance();
163        double rightDistance = rightEncoder.getDistance();
164        double averageDistance = leftDistance / 2 + rightDistance / 2;
165        return averageDistance;
166    }
167
168    /**
169     * Resets the robot's distance travelled.
170     */
171    public void resetDistance() {
172        leftEncoder.reset();
173        rightEncoder.reset();
174    }
175    private double pidOutputMagnitude;
176    private double pidOutputTurnValue;
177    private PIDSource encoderArcadeSource = new PIDSource() {
178        public double pidGet() {
179            return getTurnValue();
180        }
181    };
182    private PIDSource encoderDistSource = new PIDSource() {
183        public double pidGet() {
184            return getDistance() / MAX_INCH_PER_SECOND;
185        }
186    };
187    private PIDOutput encoderArcadeOutput = new PIDOutput() {
188        public void pidWrite(double d) {
189        }
190    };
191    private PIDOutput encoderDistOutput = new PIDOutput() {
192        public void pidWrite(double d) {
193        }
194    };
